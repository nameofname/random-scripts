def min_max(feature):
    scaled_feature = (feature - feature.min()) / (feature.max() - feature.min())
    return scaled_feature

def stand(feature):
    scaled_feature = (feature - feature.mean()) / feature.std()
    return scaled_feature

def robust(feature):
    iqr = feature.quantile(.75)-feature.quantile(.25)
    scaled_feature = (feature-feature.median()) / iqr
    return scaled_feature

def uniform(feature):
    scaled_feature = feature.rank(method='first') / (feature.shape[0]+1)
    return scaled_feature  
    
def normal(feature):
    scaled_feature = norm.ppf(uniform(feature))
    scaled_feature = pd.DataFrame(scaled_feature, columns = feature.columns)
    return scaled_feature    
    
def lognormal(feature):
    scaled_feature = lognorm.ppf(uniform(feature), 1)
    scaled_feature = pd.DataFrame(scaled_feature, columns = feature.columns)
    return scaled_feature

# Linear regression is just a dot product of X-inputs and b-coefficients.  If the model is fitted with an intercept it is added at the end.
def linear_predict(x_, b_, intercept_=0):
    return np.dot(x_, b_)+intercept_

Mathematics of Linear Regression - Loss Functions¶

Loss Functions are a parameter of the optimizer and not the linear model

In [5]:
# A common loss function for regression problems is the mean squared error of the prediction
def loss_mse(p_, y_):
    return ((p_-y_)**2).mean()

# A common fit metric for regression problems is the r-squared of the prediction
def r_squared(p_, y_):
    rss = ((p_-y_)**2).mean()
    tss = ((y_.mean()-y_)**2).mean()
    return 1 - rss/tss

# Regularization

# Lasso takes the mean squared error and adds a penalty to the absolute value of the coefficients
def loss_lasso(p_, y_, coefs, l1_=.01 ):
    loss_ = loss_mse(p_, y_)
    loss_ += np.abs(coefs).sum()*l1_
    return loss_

# Ridge takes the mean squared error and adds a penalty to the squared value of the coefficients
def loss_ridge(p_, y_, coefs, l2_=.01):
    loss_ = loss_mse(p_, y_)
    loss_ += (coefs**2).sum()*l2_
    return loss_

# Elastic net uses both L1 and L2 penalties;  Copmlete the elastic net loss function
def loss_elasticnet(p_, y_, coefs, alpha=.01, l1_ratio=.5):    
    l1_, l2_ = alpha*l1_ratio, alpha*(1-l1_ratio)    
    loss_ = loss_mse(p_, y_)
    loss_ += np.abs(coefs).sum()*l1_
    loss_ += (coefs**2).sum()*l2_
    return loss_

Mathematics of Linear Regression - Optimizers¶

The are a number of optimizers for linear regression:

  • Ordinary Least Squares: https://en.wikipedia.org/wiki/Simple_linear_regression#Fitting_the_regression_line

Other optimizers: https://scikit-learn.org/stable/modules/generated/sklearn.linear_model.Ridge.html#sklearn.linear_model.Ridge

  • svd uses a Singular Value Decomposition of X to compute the Ridge coefficients. More stable for singular matrices than cholesky.
  • cholesky uses the standard scipy.linalg.solve function to obtain a closed-form solution.
  • sparse_cg uses the conjugate gradient solver as found in scipy.sparse.linalg.cg. As an iterative algorithm, this solver is more appropriate than cholesky for large-scale data (possibility to set tol and max_iter).
  • lsqr uses the dedicated regularized least-squares routine scipy.sparse.linalg.lsqr. It is the fastest and uses an iterative procedure.
  • sag uses a Stochastic Average Gradient descent, and saga uses its improved, unbiased version named SAGA. Both methods also use an iterative procedure, and are often faster than other solvers when both n_samples and n_features are large. Note that sag and saga fast convergence is only guaranteed on features with approximately the same scale. You can preprocess the data with a scaler from sklearn.preprocessing.

def plot_bin(act, pred, bin_dim=100, lbls=['Predicted vs Actual','Bin Average','Bin #'], figsize=(10, 5)):
    plt.figure(num=None, figsize=figsize, dpi=80, facecolor='w', edgecolor='k')    
    bins = np.array_split(pred.argsort(), bin_dim)
    bins = [[pred[b].mean(), act[b].mean()] for b in bins]
    plt.plot(bins)
    plt.title(lbls[0])
    plt.ylabel(lbls[1])
    plt.xlabel(lbls[2])
    plt.legend = True
    plt.show() 
    return bins

def ols(x_, y_):
    xm, ym = x_.mean(axis=0), y_.mean()
    x_delta, y_delta = x_-xm, y_-ym
    b_ = (x_delta*y_delta[:, None]).sum(axis=0) / (x_delta*x_delta).sum()
    intercept_ = ym - (xm*b_).sum()
    return b_, intercept_

# Gradient descent common functions

# Derivative of loss_mse relative to prediction
def dloss_dp(p_, y_):
    return -2*(y_-p_)

# Derivative of prediction relative to coefficients
def dp_db(x_):
    return x_

def l1_gradient(b_, intercept_, l1_):
    grad = lambda c: l1_*((c>0)*2-1)
    return grad(b_), grad(intercept_)
    
def l2_gradient(b_, intercept_, l2_):
    grad = lambda c: 2*c*l2_
    return grad(b_), grad(intercept_)  
    

def get_gradient(x_, y_, b_, intercept_, l1_=0, l2_=0):
    # Get predictions
    p_ = linear_predict(x_, b_, intercept_)
    
    # Calculate the partial derivative of loss relative to prediction
    gradient_0 = dloss_dp(p_, y_)
    
    # Calculate the partial dervative of loss relative to coefficients
    gradient_1 = gradient_0[:, None] * dp_db(x_)
    
    # Calculate penalties    
    penalties = l1_gradient(b_, intercept_, l1_) + l2_gradient(b_, intercept_, l2_)
    return gradient_1.mean(axis=0)+penalties[0], gradient_0.mean()+penalties[1]

def find_scale(f, a=-20, b=0, n=10):    
    dim = 10
    x = np.arange(dim+1)/dim  # Get dim numbers between 0 and 1
    x = x * (b-a) + a         # Scale between a and b
    e = np.exp(x)             # Transform to exponential
    idx = np.argmin(np.array([f(e_) for e_ in e]))
    if idx == 0:
        a, b = 2*a-b, x[idx+1]
    elif idx == dim:
        a, b =  x[idx-1], 2*b-a    
    else:
        a, b = x[idx-1], x[idx+1]
    if n==0:
        return np.exp((a+b)/2)
    return find_scale(f, a, b, n-1)

# A major problem with stochastic gradient descent is optimizing the learning rate
# The following is a gradient descent optimizer that chooses an optimal learning rate for each step
def OptimalGradientDescent(x_, y_, max_iter=10000, l1=0, l2=0):
    b_, intercept_ = ols(x_, y_)    
    alpha=l1+l2    
    l1_ratio=0 if alpha==0 else l1/(l1+l2)    
    coefs = np.r_[b_, [intercept_]]
    f_loss = (
        lambda lr: 
        loss_elasticnet(
            linear_predict(X, b_-lr*b_grad, intercept_-lr*intercept_grad)
            , y, coefs, alpha, l1_ratio
        )
    )   
    learning_rt = .0001
    for i in range(max_iter):
        b_grad, intercept_grad = get_gradient(x_, y_, b_, intercept_, l1, l2)        
        log_lr = np.log(learning_rt)
        learning_rt = find_scale(f_loss, a=log_lr-5, b=log_lr+5, n=3) 
        b_ -= b_grad * learning_rt
        intercept_ -= intercept_grad * learning_rt 
        
    return b_, intercept_
